nationalavg_long <- nationalavg %>%
  pivot_longer(
    cols = -Team,          
    names_to = "Stat",
    values_to = "Value"
  )

# Plot
ggplot(nationalavg_long, aes(x = Team, y = Value, fill = Team)) +
  geom_col() +
  facet_wrap(~ Stat, scales = "free_y") +
  scale_fill_manual(values = c(
    "Lincoln Christian" = "#005596",  
    "National Average" = "grey"      
  )) +
  labs(
    title = "How Did the Crusaders Stack up Against the National Average",
    subtitle = "Stats from 2024-2025",
    x = "Team",
    y = "Value"
  ) +
  theme_minimal() +
  theme(
    plot.title = element_text(size = 20, face = "bold"),
    axis.text.x = element_text(angle = 45, hjust = 1),
    legend.position = "none"
  )

girlsbasketballstats <- girlsbasketballstats %>%
  mutate(
    FGM = as.numeric(FGM),
    FGA = as.numeric(FGA),
    PPG = as.numeric(PPG),
    FG_Efficiency = FGM / FGA 
  )

rankings <- girlsbasketballstats |>
  filter(FTM >= 15) |>                         
  group_by(Name) |>
  summarise(
    FTPercentage = mean(FTPercentage, na.rm = TRUE),
  ) |> 
  arrange(desc(FTPercentage)) |>
  slice_head(n = 10)


rankings |>
  gt() |>
  tab_header(
    title = md("**Players with the Best Free Throw Percentage**"),
    subtitle = md("*Players that made atleast 15 free throws*")
  ) |>

  cols_label(
